# Simple Login Log Analyzer
# This script checks failed and successful login attempts

log_file = "login.log"

failed_logins = 0
successful_logins = 0
suspicious_ips = {}

# Open the log file
with open(log_file, "r") as file:
    logs = file.readlines()

# Check every line in the log file
for line in logs:

    if "Failed password" in line:
        failed_logins += 1

        # Split the line into words
        words = line.split()

        # Get IP address after "from"
        if "from" in words:
            ip_position = words.index("from")
            ip = words[ip_position + 1]

            if ip in suspicious_ips:
                suspicious_ips[ip] += 1
            else:
                suspicious_ips[ip] = 1

    elif "Accepted password" in line:
        successful_logins += 1


# Show results
print("=== LOGIN LOG ANALYSIS ===")

print("Failed logins:", failed_logins)
print("Successful logins:", successful_logins)

print("\nSuspicious IP addresses:")

for ip in suspicious_ips:

    attempts = suspicious_ips[ip]

    if attempts >= 3:
        print(ip, "-", attempts, "failed attempts")
        print("Possible brute force attack!")
