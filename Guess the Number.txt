import random

secret_number = random.randint(1, 10)

attempts = 0

print("Guess the number from 1 to 10")

while True:

    guess = int(input("Your number: "))
    attempts += 1

    if guess < secret_number:
        print("Too low")

    elif guess > secret_number:
        print("Too high")

    else:
        print("You win!")
        print("Attempts:", attempts)
        break
